install -d 2011-08-22
cd  2011-08-22
wget
 * http://ftp.osuosl.org/pub/funtoo/gentoo-stable/x86-64bit/generic_64/2011-08-22/stage3-generic_64-gentoo-stable-2011-08-22.tar.xz
